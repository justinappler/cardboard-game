001    package com.immortallabs.cardboard.ui;
002    
003    import java.awt.Graphics;
004    import java.awt.Graphics2D;
005    import java.awt.Point;
006    import java.awt.RenderingHints;
007    import java.awt.image.BufferedImage;
008    
009    import javax.swing.JComponent;
010    
011    import com.immortallabs.cardboard.CardBoardResource;
012    import com.immortallabs.cardboard.game.Card;
013    import com.immortallabs.cardboard.game.SetsBoard;
014    
015    import edu.profdalbey.Natural;
016    
017    /**
018     * The CardGraphic class is used to draw the Cardboard game cards in the
019     * GameWindow. It does this by extending Component and overwriting the paint
020     * method. This class also contains methods to animate and scale the cards
021     * appropriately. <br />
022     * <br />
023     * <b>Operations:</b>
024     * <ul>
025     * <li>The CardGraphic class can create a card for the game board.
026     * <li>The CardGraphic class can animate itself card.
027     * <li>The CardGraphic class can scale the size of itself to fit on the game
028     * board.
029     * </ul>
030     * 
031     * @author Brad Barrows
032     * @version 1.3
033     * @version Jan. 3rd, 2008
034     * @history - 1/13/09 1.3 Updated packages and tags <br> - 12/04/08 1.2 More
035     *          Pseudocoding<br> - 12/02/08 1.1 Added Version History & Pseudocode<br>
036     */
037    
038    public class CardGraphic extends JComponent 
039    {
040    
041        /**
042         * Contains the card that this graphic represents
043         */
044        private Card card;
045        /**
046         * Contains the Image of the card graphic
047         */
048        private BufferedImage cardGraphic;
049        /**
050         * A reference to the instance of the ActionMap
051         */
052        private ActionMap actionMapInst;
053        /**
054         * The point that the card graphic is being drawn at
055         */
056        private Point drawLocation;
057        /**
058         * The row this card is in
059         */
060        private Natural row;
061        /**
062         * The column this card is in
063         */
064        private Natural column;
065        /**
066         * A reference to the instance of the GameWindow
067         */
068        private GameWindow gamewindow;
069        /**
070         * The number of rows of cards on the game board
071         */
072        private SetsBoard.ROWS numRows;
073        /**
074         * The width of the cards at each scale size
075         */
076        private static final int[] kCardWidth = {150, 150, 130, 110};
077    
078        /**
079         * The height of the cards at each scale size
080         */
081        private static final int[] kCardHeight = {100, 100, 80, 60};
082    
083    
084    
085        /**
086         * Creates a CardGraphic to represent a card at the position row, column in
087         * the game board.
088         * 
089         * @param card
090         *            the card to represent
091         * @param actionmap
092         *            the ActionMap instance to use
093         * @param row
094         *            the row of the gameboard to place the card
095         * @param column
096         *            the column of the gameboard to place the card
097         * @param gamewindow
098         *            the GameWindow instance to use
099         * @param drawLocation
100         *            the location to draw the card at            
101         * @see Card
102         * @see ActionMap
103         * @see GameWindow
104         * @see Natural
105         */
106        public CardGraphic(Card card, ActionMap actionmap, Natural row,
107                Natural column, GameWindow gamewindow, Point drawLocation)
108        {
109            // SET this card
110            this.card = card;
111            // SET this ActionMap instance
112            this.actionMapInst = actionmap;
113            // SET this row
114            this.row = row;
115            // SET this column
116            this.column = column;
117            // SET this GameWindow instance
118            this.gamewindow = gamewindow;
119            // SET the number of rows to 4
120            this.numRows = SetsBoard.ROWS.FOUR;
121    
122            // SET drawLocation to the point of the board where this card will be
123            // drawn 
124            this.drawLocation = drawLocation;
125    
126            // LOAD the current theme into theme
127    
128            // LOAD the card graphic into cardGraphic
129            String fileName =
130                "" + card.getShape().ordinal() + card.getColor().ordinal()
131                        + card.getFill().ordinal()
132                        + card.getNumber().ordinal() + ".jpg";
133    
134            cardGraphic = CardBoardResource.getImage(this, fileName);
135        }
136        
137        /**
138         * Returns the point at which this card is to be drawn.
139         * @return the Point to draw this card at
140         */
141        public Point getDrawLocation()
142        {
143            return drawLocation;
144        }
145    
146    
147        /**
148         * Scales the size of this CardGraphic to fit on the gameboard with 4,5,6 or
149         * 7 rows. This method is called when the number of rows of cards on the
150         * gameboard increases. It is used to scale the size of the cards to all fit
151         * in the GameWindow.
152         * 
153         * @param rows
154         *            the number of rows of cards on the gameboard
155         * @see GameWindow
156         */
157        public void scaleTo(SetsBoard.ROWS rows)
158        {
159            // SCALE cardGraphic to the appropriate WIDTH and HEIGHT for the number
160            // of rows on the game board
161            int width = kCardWidth[rows.ordinal()];
162            int height = kCardHeight[rows.ordinal()];
163            BufferedImage scaledImage =
164                    new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
165            // Paint scaled version of image to new image
166            Graphics2D graphics2D = scaledImage.createGraphics();
167            graphics2D.setRenderingHint(RenderingHints.KEY_INTERPOLATION,
168                RenderingHints.VALUE_INTERPOLATION_BILINEAR);
169            graphics2D.drawImage(cardGraphic, 0, 0, width, height, null);
170            // clean up
171            graphics2D.dispose();
172    
173            cardGraphic = scaledImage;
174    
175        }
176    
177        /**
178         * Returns the card.
179         * 
180         * @return The card
181         */
182        public Card getCard()
183        {
184            // RETURN card
185            return card;
186        }
187    
188        @Override
189        public void paint(Graphics g)
190        {
191            // PAINT cardGraphic to g at drawLocation
192            g.drawImage(cardGraphic, drawLocation.x,
193                drawLocation.y, this);
194        }
195    
196    }
