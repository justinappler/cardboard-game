001    package com.immortallabs.cardboard.ui;
002    
003    import java.awt.Graphics;
004    import java.awt.Point;
005    import java.awt.image.BufferedImage;
006    
007    import javax.swing.JComponent;
008    
009    import com.immortallabs.cardboard.CardBoardResource;
010    
011    /**
012     * The Button class is used to draw one of the three Cardboard game buttons in
013     * the GameWindow. The type of button to be created is defined in the Button
014     * constructor using the ButtonType enumeration. <br />
015     * <br />
016     * <b>Operations:</b>
017     * <ul>
018     * <li>The Button class can create three types of buttons for the game board.
019     * </ul>
020     * 
021     * @author Nikhil Kowshik
022     * @author Formerly Brad Barrows at version 1.2
023     * @version 1.3
024     * @version 1/14/09 Version History
025     * @history - 12/04/08 1.2 More Pseudocoding<br> - 12/02/08 1.1 Added Version
026     *          History & Pseudocode <br>
027     */
028    
029    public class CardBoardButton extends JComponent
030    {
031       /**
032        * Contains the Image of the button when it is not being pressed
033        */
034       private BufferedImage buttonUp;
035    
036       /**
037        * The point on the GameWindow where this button is drawn
038        */
039       private Point drawLocation;
040       /**
041        * The current status of the button
042        */
043       private BUTTONSTATUS buttonStatus;
044       /**
045        * Contains the message to be printed on the button
046        */
047       private String message;
048    
049       /**
050        * The ButtonType enumeration is used to describe what type of button is to
051        * be created.
052        */
053       private enum BUTTONSTATUS
054       {
055          /** The button is up * */
056          UP,
057          /** The button is down * */
058          DOWN
059       }
060    
061       /**
062        * The type of button
063        */
064       private BUTTONTYPE type;
065    
066       /**
067        * The ButtonType enumeration is used to describe what type of button is to
068        * be created.
069        */
070       public enum BUTTONTYPE
071       {
072          /** A hint button * */
073          HINT,
074          /** A three more button * */
075          THREEMORE,
076          /** A pause button * */
077          PAUSE,
078          /** An unpause button * */
079          UNPAUSE,
080          /** A welcome button * */
081          WELCOME,
082          /** A explain button * */
083          EXPLAIN
084          /** An OK button for dialog boxes */
085          , OK
086       }
087    
088       /**
089        * Creates one of the possible buttons at a certain point in the GameWindow.
090        * 
091        * @param pt
092        *           where to draw the Button
093        * @param actionmap
094        *           the instance of ActionMap to use
095        * @param type
096        *           the type of button to create
097        * @param message
098        *           the message to place on the button
099        * @see ActionMap
100        */
101       public CardBoardButton(Point pt, ActionMap actionmap, BUTTONTYPE type,
102             String message)
103       {
104          // CREATE a new string for the buttonUp filename
105          String up = null;
106    
107          // IF type is a HINT BUTTON
108          if (type == BUTTONTYPE.HINT)
109          {
110             // SET up to current theme's hint button up image filename
111             up = "hint.jpg";
112          }
113          // ELSE IF type is a THREE MORE BUTTON
114          else if (type == BUTTONTYPE.THREEMORE)
115          {
116             // SET up to current theme's three more button up image filename
117             up = "threeMore.jpg";
118          }
119          // ELSE IF type is a PAUSE BUTTON
120          else if (type == BUTTONTYPE.PAUSE)
121          {
122             // SET up to current theme's pause button up image filename
123             up = "pause.jpg";
124          }
125          // ELSE IF type is a UNPAUSE BUTTON
126          else if (type == BUTTONTYPE.UNPAUSE)
127          {
128             up = "resume.jpg";
129          }
130          // ELSE IF type is a WELCOME BUTTON
131          else if (type == BUTTONTYPE.WELCOME)
132          {
133             up = "start.jpg";
134          }
135          // ELSE IF type is a WELCOME BUTTON
136          else if (type == BUTTONTYPE.EXPLAIN)
137          {
138             up = "explain.jpg";
139          }
140          else if (type == BUTTONTYPE.OK)
141          {
142             up = "ok.jpg";
143          }
144    
145          buttonUp = CardBoardResource.getImage(this, up);
146    
147          // ADD actionmap as this components mouseListener
148          this.addMouseListener(actionmap);
149    
150          // SET this class's drawLocation to pt
151          drawLocation = pt;
152    
153          // SET this class's to type
154          this.type = type;
155    
156          // SET this class's to message
157          this.message = message;
158    
159          // SET this class's buttonStatus to BUTTONSTATUS.UP
160          buttonStatus = BUTTONSTATUS.UP;
161       }
162    
163       /**
164        * Returns the type of button.
165        * 
166        * @return The type of button
167        */
168       public BUTTONTYPE getType()
169       {
170          // RETURN type
171          return type;
172       }
173    
174       /**
175        * Draw the buttons onto the main window.
176        * 
177        * @param g
178        *           the graphics context to draw the buttons onto
179        */
180       public void paint(Graphics g)
181       {
182          // PAINT buttonUp image to g at drawLocation
183          g.drawImage(buttonUp, drawLocation.x, drawLocation.y, null);
184       }
185    }
