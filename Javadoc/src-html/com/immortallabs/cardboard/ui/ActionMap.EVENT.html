456        /**
457         * Instantiates an ActionMap that will listen for actions from the UI or the
458         * keyboard. It will use GameState to get the keyboard bindings and send
459         * actions.
460         * 
461         * @param gameGameState
462         *            The current gameState of the CardBoard game.
463         * @param gameWindow
464         *            The current gameWindow displaying the game
465         */
466        public ActionMap(GameState gameGameState, GameWindow gameWindow)
467        {
468            /** Get the bindings for all the keys */
469            KeyBindings keyBind;
470            // SET gameState to gameGameState
471            this.gameState = gameGameState;
472            // SET this.gameWindow to gameWidow
473            this.gameWindow = gameWindow;
474            // SET noAction to false
475            noAction = false;
476            // SET keyMap to a new hash map of strings that represent a key's code
477            // and the corresponding event.
478            keyMap = new HashMap<String, EVENT>();
479            keyBind = new KeyBindings();
480            // FOR each keyBinding in KeyBindings.KEYBINDS
481            for (KeyBindings.KEY_BIND key : KeyBindings.KEY_BIND.values())
482            {
483                // ADD the results of getKeyBinding with keyBindins and event to
484                keyMap.put(keyBind.getKeyBinding(key), EVENT.valueOf(key.name()));
485                // END FOR
486            }
487        }

489        /**
490         * Processes an event when a key is typed from the keyboard.
491         * 
492         * @param interEvent
493         *            The event to be processed
494         */
495        public void processEvent(EVENT interEvent)
496        {
497            try
498            {
499                // IF interEvent equals PAUSE
500                if (interEvent.equals(EVENT.PAUSE) && !noAction)
501                {
502                    // SET noAction to true
503                    noAction = true;
504                    // CALL GameWIndow's showPauseDisplay
505                    gameWindow.showButton(CardBoardButton.BUTTONTYPE.PAUSE);
506                    // CALL GameState's processAction with interEvent's getAction
507                    gameState.processAction(interEvent.getAction());
508                    // ELSE IF interEvent equals UNPAUSE
509                }
510                else if (interEvent.equals(EVENT.UNPAUSE) && noAction)
511                {
512                    // SET noAction to false
513                    noAction = false;
514                    // CALL GameWindow's removePauseDisplay
515                    gameWindow.removeButton(CardBoardButton.BUTTONTYPE.UNPAUSE);
516                    // CALL GameState's processAction with interEvent's getAction
517                    gameState.processAction(interEvent.getAction());
518                    // ELSE IF noAction = false
519                }
520                else if (interEvent.equals(EVENT.HINT))
521                {
522                    // GET the setsBoard from gameWindow
523                    SetsBoard setsBoard = gameState.getBoard();
524                    // GET the inlay size before hint is called
525                    int before = setsBoard.getInlaySize().intValue();
526                    // Move the selector for a hint
527                    gameWindow.moveSelector(GameWindow.DIRECTION.HINTDIR);
528                    // CALL processAction with ACTION's HINT_SELECTED
529                    gameState.processAction(GameState.ACTION.HINT);
530                    // IF the hint is not the third hint
531                    if (before != 2 && !gameWindow.hasMessage)
532                        // repaint gameWindow
533                        gameWindow.repaint();
534                }
535                else if (!noAction)
536                {
537                    // CALL GameState's processAction with interEvent's getAction
538                    gameState.processAction(interEvent.getAction());
539                    // END IF
540                }
541            }
542            catch (NoActionException e)
543            {
544                e.printStackTrace();
545            }
546            if (interEvent.equals(EVENT.REMOVE_INLAY_POS_ONE)
547                    || interEvent.equals(EVENT.REMOVE_INLAY_POS_TWO))
548                gameWindow.repaint();
549        }

551        /**
552         * Processes a button when it is clicked.
553         * 
554         * @param clickedButton
555         *            The button that is being processed
556         */
557        public void processEvent(CardBoardButton clickedButton)
558        {
559            // IF clickedButton's getType equals BUTTONTYPE's WELCOME
560            if (clickedButton.getType() == CardBoardButton.BUTTONTYPE.WELCOME)
561            {
562                // CALL GameWindow's removeButton with WELCOME
563                gameWindow.removeButton(CardBoardButton.BUTTONTYPE.WELCOME);
564                // CALL GameState;s processAction with Action's UNPAUSE
565                gameState.processAction(GameState.ACTION.UNPAUSE);
566                // END IF
567                gameWindow.repaint();
568            }
569            // ELSE IF clickedButton's getType equals BUTTONTYPE's OK
570            if (clickedButton.getType() == CardBoardButton.BUTTONTYPE.OK)
571            {
572                gameWindow.repaint();
573            }
574            // ELSE IF clickedButton's getType equals BUTTONTYPE's PAUSE
575            else if (clickedButton.getType() == CardBoardButton.BUTTONTYPE.PAUSE
576                    && !noAction)
577            {
578                // SET noAction to true
579                noAction = true;
580                // CALL GameWIndow's showPauseDisplay
581                gameWindow.showButton(CardBoardButton.BUTTONTYPE.PAUSE);
582                // CALL GameState's processAction with event's getAction
583                gameState.processAction(GameState.ACTION.PAUSE);
584                gameWindow.repaint();
585            }
586            // ELSE IF clickedButton's getType equals BUTTONTYPE's UNPAUSE
587            else if (clickedButton.getType() == CardBoardButton.BUTTONTYPE.UNPAUSE
588                    && noAction)
589            {
590                // SET noAction to false
591                noAction = false;
592                // CALL GameWindow's removePauseDisplay
593                gameWindow.removeButton(CardBoardButton.BUTTONTYPE.UNPAUSE);
594                // CALL GameState's processAction with event's getAction
595                gameState.processAction(GameState.ACTION.UNPAUSE);
596                gameWindow.repaint();
597            }
598            // ELSE IF noAction should not be taken
599            else if (!noAction)
600            {
601                // IF clickedButton's getType equals BUTTONTYPE's HINT
602                if (clickedButton.getType() == CardBoardButton.BUTTONTYPE.HINT)
603                {
604                    // GET the setsBoard from gameWindow
605                    SetsBoard setsBoard = gameState.getBoard();
606                    // GET the inlay size before hint is called
607                    int before = setsBoard.getInlaySize().intValue();
608                    // Move the selector for a hint
609                    gameWindow.moveSelector(GameWindow.DIRECTION.HINTDIR);
610                    // CALL processAction with ACTION's HINT_SELECTED
611                    gameState.processAction(GameState.ACTION.HINT);
612                    // IF the hint is not the third hint
613                    if (before != 2 && !gameWindow.hasMessage)
614                        // repaint gameWindow
615                        gameWindow.repaint();
616                }
617                // ELSE IF clickedButton's getType equals BUTTONTYPE's THREEMORE
618                else if (clickedButton.getType() == CardBoardButton.BUTTONTYPE.THREEMORE)
619                {
620                    // CALL GameState's processAction with ACTION's
621                    // THREE_MORE_SELECTED
622                    gameState.processAction(GameState.ACTION.THREE_MORE);
623                }
624                // ELSE IF clickedButton's getType equals BUTTONTYPE's WHY
625                else if (clickedButton.getType() == CardBoardButton.BUTTONTYPE.EXPLAIN)
626                {
627                    // CALL GameState's processAction with ACTION's EXPLANATION
628                    gameState.processAction(GameState.ACTION.EXPLANATION);
629                    // set noAction to true
630                    noAction = true;
631                }
632                // END IF
633            }
634        }

673        /**
674         * Handles when a key is pressed
675         * 
676         * @param keyEvent
677         *            The key that is pressed
678         */
679        public void keyPressed(KeyEvent keyEvent)
680        {
681            // CREATE an EVENT interEvent;
682            EVENT interEvent;
683            // SET interEvent to keyMap's get with keyEvent's keyText with
684            // keyEvent's keyCode
685            interEvent = keyMap.get(new Integer(keyEvent.getKeyCode()).toString());
686            //GIVE focus to actionMap
687            this.requestFocusInWindow();
688            //IF key is an alt
689            if (keyEvent.equals(KeyEvent.VK_ALT))
690            {
691                //FOCUS on menu
692                gameWindow.menu.getMenuBar().requestFocusInWindow();
693            }
694            // ELSE IF interEvent doesn't equal null
695            else if (interEvent != null)
696            {
697                // IF interEvent equals EVENT's SELECT
698                if (interEvent == EVENT.SELECT)
699                {
700                    // CALL processEvent with gameWindow's getSelected
701                    processEvent(gameWindow.getSelectedCardGraphic());
702                    // ELSE IF interEvent equals an EVENT direction
703                }
704                else if (EnumSet.of(EVENT.LEFT, EVENT.RIGHT, EVENT.UP, EVENT.DOWN)
705                        .contains(interEvent))
706                {
707                    // CALL GameWindow's moveSelector on GameWindow's DIRECTION
708                    gameWindow.moveSelector(GameWindow.DIRECTION.valueOf(interEvent
709                            .name()));
710                }
711                else
712                {
713                    // CALL processEvent on interEvent's getAction
714                    processEvent(interEvent);
715                    // END IF
716                }
717            }
718        }

720        public void mousePressed(MouseEvent mouseEvent)
721        {
722            Object clickedObject =
723                    gameWindow.getSource(mouseEvent.getX(), mouseEvent.getY());
724            // IF Object's getClass with mouseEvent's getSource is a Button
725            if (clickedObject.getClass() == CardBoardButton.class)
726            {
727                // call processEvent with Object's getClass with mouseEvent's
728                // getSource
729                processEvent((CardBoardButton) clickedObject);
730                // ELSE IF Object's getClass with mouseEvent's getSource is a Card
731            }
732            else if (clickedObject.getClass() == CardGraphic.class)
733            {
734                // call processEvent with Object's getClass with mouseEvent's
735                // getSource
736                processEvent((CardGraphic) clickedObject);
737                // END IF
738            }
739        }

741        public void actionPerformed(ActionEvent actionEvent)
742        {
743            // Get the current instance of CardBoardPreference
744            CardBoardPreferences pref = CardBoardPreferences.getInstance();
745            // If the action is SOUND_TOGGLE than toggle the sound.
746            if (actionEvent.getActionCommand().equals(EVENT.SOUND_TOGGLE.name()))
747            {
748                // If the sound is currently turned off than turn it on
749                if (pref.getPreference(CardBoardPreferences.PREF.SOUNDS).equals(
750                    "OFF"))
751                {
752                    // SET the sound preference to on
753                    pref.setPreference(CardBoardPreferences.PREF.SOUNDS, "ON");
754                }
755                // ELSE turn it off
756                else
757                {
758                    // SET the sound preference to off
759                    pref.setPreference(CardBoardPreferences.PREF.SOUNDS, "OFF");
760                }
761            }
762            // ElSE key bindings is selected
763            else if (actionEvent.getActionCommand().equals(EVENT.KEY_BINDS.name()))
764            {
765                Object[][] data =
766                        new Object[][] {
767                                { "Hint", "h" },
768                                { "Pausing the game", "p" },
769                                { "Asking for three more cards", "t" },
770                                { "Moving a seleced card to the inlay", "enter" },
771                                { "Moving the selected card up", "up arrow" },
772                                { "Moving the selected card down", "down arrow" },
773                                { "Moving the selected card left", "left arrow" },
774                                { "Moving the selected card right", "right arrow" },
775                                { "Moving inlay card 1 back to the table", "1" },
776                                { "Moving inlay card 2 back to the table", "2" },
777                                { "Player one to buzz in", "left shift" },
778                                { "Player two to buzz in", "space" },
779                                { "Player three to buzz in", "right control" },
780                                { "Player four to buzz in", "numpad" }, };
781                Object[] columns = new Object[] { "Action", "Key Bind" };
782                DefaultTableModel model = new DefaultTableModel(data, columns);
783                JTable table = new JTable();
784                table.setModel(model);
785    
786                JScrollPane jsp = new JScrollPane();
787                jsp.setViewportView(table);
788    
789                JOptionPane.showMessageDialog(gameWindow, jsp, "Key Bindings",
790                    JOptionPane.INFORMATION_MESSAGE, new ImageIcon());
791            }
792            // ElSE if the action is to change the number of players
793            else if (playerSet.contains(EVENT.valueOf(actionEvent
794                    .getActionCommand())))
795            {
796                // SET the current preference to that number of players
797                pref.setPreference(CardBoardPreferences.PREF.PLAYERS, actionEvent
798                        .getActionCommand());
799                // CALL process action new multiplayer game
800                gameState.processAction(GameState.ACTION.NEW_MULTIPLAYER_GAME);
801            }
802            // ELSE IF the action is to change the amount of time
803            else if (timeSet
804                    .contains(EVENT.valueOf(actionEvent.getActionCommand())))
805            {
806                // SET the current preferences to new time limit
807                pref.setPreference(CardBoardPreferences.PREF.TIME_LIMIT,
808                    actionEvent.getActionCommand());
809            }
810            // ELSE IF the action is to show the about page
811            else if (actionEvent.getActionCommand().equals(EVENT.ABOUT.name()))
812            {
813                gameWindow
814                        .showPage("http://wiki.csc.calpoly.edu/ImmortalLabs/wiki/about");
815            }
816            // ELSE IF the action is to show the how to page
817            else if (actionEvent.getActionCommand().equals(EVENT.HOW_TO.name()))
818            {
819                gameWindow
820                        .showPage("http://wiki.csc.calpoly.edu/ImmortalLabs/wiki/rules");
821            }
822            // ELSE send the action to processAction
823            else
824            {
825                // FOR each event in EVENT
826                for (EVENT event : EVENT.values())
827                {
828                    // IF actionEvent's getActionCommand equals EVENT's name
829                    if (actionEvent.getActionCommand() == event.name())
830                    {
831                        try
832                        {
833                            // CALL GameState's processAction with EVENT's getAction
834                            gameState.processAction(event.getAction());
835                        }
836                        catch (NoActionException e)
837                        {
838                            e.printStackTrace();
839                        }
840                        // END IF
841                    }
842                    // END FOR
843                }
844            }
845            // IF the the action is to start a new game and game is paused
846            if (gameModes.contains(EVENT.valueOf(actionEvent.getActionCommand())))
847            {
848                // set actionMap to allow actions
849                noAction = false;
850                // remove the pause screen from game window-
851                gameWindow.removeButton(CardBoardButton.BUTTONTYPE.UNPAUSE);
852    
853                // Get rid of multiplayer score if applicable
854                gameWindow.displayBuzzedPlayer(null);
855            }
856            // Repaint the window in case the action changed anything on the screen
857            gameWindow.repaint();
858        }
