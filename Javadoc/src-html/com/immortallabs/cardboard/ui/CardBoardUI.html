001    package com.immortallabs.cardboard.ui;
002    
003    import com.immortallabs.cardboard.game.Card;
004    import com.immortallabs.cardboard.game.CardBoardMessage;
005    import com.immortallabs.cardboard.game.GameState;
006    
007    /**
008     * <p>
009     * The CardBoardUI class is an interface that must implemented by all user
010     * interfaces for the Cardboard Sets Game.
011     * </p>
012     * <b>Operations:</b>
013     * <ul>
014     * <li>A class that implements CardBoardUI can provide the user with the
015     * Player's Name
016     * <li>A class that implements CardBoardUI can provide the user with the
017     * currently selected card.
018     * </ul>
019     * 
020     * @author Justin C. Appler
021     * @author Immortal Labs CSC309 W09
022     * @version 1.2
023     * @version January 12th, 2009
024     * @history - 01/12/09 1.2 Moved the class to the new project<br>
025     *          - 12/04/08 1.1 Added the getSelected and showExplanation method<br>
026     *          - 12/01/08 1.0 Initial Class Skeleton<br>
027     */
028    public interface CardBoardUI
029    {
030        /**
031         * Retrieves the name/handle of the current player from the User Interface.
032         * 
033         * @return The name or handle of the player who received the high score
034         */
035        String getName();
036    
037        /**
038         * Gets the Card currently selected on the game board.
039         * 
040         * @return The Card currently selected on the game board.
041         */
042        Card getSelected();
043    
044        /**
045         * Shows the user a message regarding an action they've taken
046         * 
047         * @param message
048         *          The message to be shown to the user
049         */
050        void showMessage(CardBoardMessage message);
051    
052        /**
053         * Displays the player that is currently buzzed in
054         * @param player
055         *    The player that is currently buzzed in
056         *    null if no player is currently buzzed in
057         */
058        void displayBuzzedPlayer(GameState.PLAYER player);
059        
060        /**
061         * Redraws or refreshes the user interface after an unexposed
062         * change in the GameState
063         */
064        void redraw();
065    }
