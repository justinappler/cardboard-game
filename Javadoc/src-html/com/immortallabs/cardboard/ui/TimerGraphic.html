001    package com.immortallabs.cardboard.ui;
002    
003    import java.awt.Graphics;
004    import java.awt.Point;
005    
006    import javax.swing.JComponent;
007    
008    import com.immortallabs.cardboard.game.Timer;
009    
010    import edu.profdalbey.Natural;
011    
012    /**
013     * <p>
014     * The TimerGraphic class displays the information of a Timer in a graphical
015     * display.
016     * </p>
017     * <br>
018     * <br>
019     * <b>Operations:</b>
020     * <ul>
021     * <li>The TimerGraphic class can graphically display a Timer.
022     * </ul>
023     * 
024     * @author Ryan S. Lange
025     * @version 0.3
026     * @version 12/04/08
027     * @see Timer
028     * @see Point
029     * @history - 01/21/09 0.3 Added source code - 12/04/08 0.2 Added Version
030     *          History & Pseudocode<br> - 12/01/08 0.1 Initial Class Skeleton<br>
031     */
032    
033    public class TimerGraphic extends JComponent
034    {
035        /**
036         * Instance of the Timer
037         */
038        private Timer theTimer;
039    
040        /**
041         * Holds the total amount of seconds on the timer
042         */
043        private Natural totalSeconds;
044    
045        /**
046         * Holds the amount of hours the seconds has
047         */
048        private int hours;
049    
050        /**
051         * Holds the amount of minutes the seconds has
052         */
053        private int minutes;
054    
055        /**
056         * Holds the amount of seconds from the TotalSeconds
057         */
058        private int seconds;
059    
060        /**
061         * Holds the string to show the time
062         */
063        private String timeString;
064    
065        /**
066         * Holds the point to be painted
067         */
068        private Point myPoint;
069    
070        /**
071         * Final for 10
072         */
073        private static final int kTen = 10;
074    
075        /**
076         * Final for 60
077         */
078        private static final int kSixty = 60;
079    
080        /**
081         * Final for 26
082         */
083        private static final int kTwentySix = 26;
084    
085        /**
086         * Final for 130
087         */
088        private static final int kOneThirty = 130;
089    
090        /**
091         * Final for 30
092         */
093        private static final int kThirty = 30;
094    
095        /**
096         * Initializes the TimerGraphic with a Timer t at the position of the Point
097         * p.
098         * 
099         * @param p
100         *            the Point at which to draw the TimerGraphic window.
101         * @param t
102         *            the Timer that we are displaying.
103         */
104        public TimerGraphic(Point p, Timer t)
105        {
106            // SET thetimer to t
107            theTimer = t;
108    
109            // SET timeString to null
110            timeString = null;
111    
112            // SET myPoint to p
113            myPoint = p;
114    
115            // CALL makeTimerString
116            makeTimerString();
117        }
118    
119        /**
120         * Makes the string to put into the text field of the window
121         */
122        private String makeTimerString()
123        {
124            // SET totalSeconds to return value of theTimer's seconds
125            totalSeconds = theTimer.getSeconds();
126    
127            // CALL convertTotalSeconds
128            convertTotalSeconds();
129    
130            // SET theHour, theMin, and theSec = null
131            String theHour = null, theMin = null, theSec = null;
132    
133            // SET theHour to string of hours
134            theHour = String.valueOf(hours);
135    
136            // IF minutes are less than 10
137            if (minutes < kTen)
138            {
139                // theMin = "0" + string of minutes
140                theMin = "0" + String.valueOf(minutes);
141            }
142            // ELSE
143            else
144            {
145                // theMin = string of minutes
146                theMin = String.valueOf(minutes);
147            }
148            // END IF
149    
150            // IF seconds less than 10
151            if (seconds < kTen)
152            {
153                // theSec = "0" + string of seconds
154                theSec = "0" + String.valueOf(seconds);
155            }
156            // ELSE
157            else
158            {
159                // theSec = string of seconds
160                theSec = String.valueOf(seconds);
161            }
162            // END IF
163    
164            // timeString = theHour + ":" + theMin + ":" + theSec
165            timeString = theHour + ":" + theMin + ":" + theSec;
166    
167            // RETURN timeString
168            return timeString;
169        }
170    
171        /**
172         * Converts the totalSeconds into hours, minutes, and seconds
173         */
174        private void convertTotalSeconds()
175        {
176            // SET hours to totalSeconds / (60 * 60)
177            hours = totalSeconds.intValue() / (kSixty * kSixty);
178    
179            // SET minutes to (totalSeconds / 60) % 60
180            minutes = (totalSeconds.intValue() / kSixty) % kSixty;
181    
182            // SET seconds to totalSeconds % 60
183            seconds = totalSeconds.intValue() % kSixty;
184        }
185    
186        @Override
187        public void paint(Graphics g)
188        {
189            // PAINT the timer graphics window
190            // super.paint(g);
191    
192            // SET color of background to black
193            g.setColor((java.awt.Color.black));
194    
195            // FILL rect with x and y
196            g.fillRect(myPoint.x, myPoint.y - kTwentySix, kOneThirty, kThirty);
197    
198            // SET color of text to white
199            g.setColor(java.awt.Color.WHITE);
200    
201            // SET font to arial bold size 32
202            g.setFont(java.awt.Font.decode("Arial-BOLD-24"));
203    
204            // CALL drawString with parameters return of makeTimerString, x, and y
205            g.drawString(makeTimerString(), myPoint.x, myPoint.y);
206    
207            // CALL drawString with parameters timeString, x, an dy
208            g.drawString(timeString, myPoint.x, myPoint.y);
209        }
210    }
