006    /**
007     * <p>
008     * The KeyBindings class models all key binding for the game. If a player
009     * changes any key bindings, the class persist it to retain the information for
010     * future games.
011     * </p>
012     * <b>Operations:</b>
013     * <ul>
014     * <li>The KeyBindings class can get a specific key binding for an action
015     * </ul>
016     * 
017     * @author Nikhil Kowshik
018     * @author Immortal Labs CSC309 W09

022     * @version 12/05/08
023     * @history - 12/03/08 Version 1.1 Added Version History. Moved all KeyBinding
024     *          methods and enums to KeyBindings class. Added TODO - 12/05/08
025     *          Version 1.2 Finished peusdocode and TODO
026     */
027    public class KeyBindings
028    {
029       /**
030        * Key for using a hint: 'H' key on the keyboard
031        */
032       private static final String kKeyHint = "72";
033    
034       /**
035        * Key for pausing the game: 'P' key on the keyboard
036        */
037       private static final String kKeyPause = "80";
038    
039       /**
040        * Key for pausing the game: 'R' key on the keyboard
041        */
042       private static final String kKeyUnpause = "82";
043    
044       /**
045        * Key for asking for three more cards: 'T' key on the keyboard
046        */
047       private static final String kKeyThreeMore = "84";
048    
049       /**
050        * Key for moving a card from the table to the inlay: ENTER key on the
051        * keyboard
052        */
053       private static final String kKeySelect = "10";
054    
055       /**
056        * Key for moving the selected card up: UP arrow key on the keyboard
057        */
058       private static final String kKeyUp = "38";
059    
060       /**
061        * Key for moving the selected card down: DOWN arrow key on the keyboard
062        */
063       private static final String kKeyDown = "40";
064    
065       /**
066        * Key for moving the selected card left: LEFT arrow key on the keyboard
067        */
068       private static final String kKeyLeft = "37";
069    
070       /**
071        * Key for moving the selected card right: RIGHT arrow key on the keyboard
072        */
073       private static final String kKeyRight = "39";
074    
075       /**
076        * Key for moving the first card in the inlay back to the table: '1' key on
077        * the keyboard
078        */
079       private static final String kKeyRemoveInlayOne = "49";
080    
081       /**
082        * Key for moving the second card in the inlay back to the table: '2' key on
083        * the keyboard
084        */
085       private static final String kKeyRemoveInlayTwo = "50";
086    
087       /**
088        * Key for player one to buzz in: LEFT SHIFT key on the keyboard
089        */
090       private static final String kKeyBuzzOne = "16";
091    
092       /**
093        * Key for player two to buzz in: SPACE key on the keyboard
094        */
095       private static final String kKeyBuzzTwo = "32";
096    
097       /**
098        * Key for player three to buzz in: RIGHT CONTROL key on the keyboard
099        */
100       private static final String kKeyBuzzThree = "17";
101    
102       /**
103        * Key for player four to buzz in: NUMPAD 0 key on the keyboard
104        */
105       private static final String kKeyBuzzFour = "47";
106    
107       /**
108        * Holds all the key bindings with their current values.
109        */
110       private HashMap<KEY_BIND, String> keyBindings;
111    
112       /**
113        * KEY_BIND models all the different keyboard bindings for all the different
114        * actions.
115        */
116       public enum KEY_BIND
117       {
118          /** The keyboard key that is binded to the Hint Button */
119          HINT,
120    
121          /** The keyboard key that is binded to the Pause Button */
122          PAUSE,
123    
124          /** The keyboard key that unpauses the game; it is the same key as pause */
125          UNPAUSE,
126    
127          /** The keyboard key that is binded to the 3 More Button */
128          THREE_MORE,
129    
130          /** The keyboard key that chooses the selected card */
131          SELECT,
132    
133          /** The keyboard key that moves the selected card up */
134          UP,
135    
136          /** The keyboard key that moves the selected card down */
137          DOWN,
138    
139          /** The keyboard key that moves the selected card left */
140          LEFT,
141    
142          /** The keyboard key that moves the selected card right */
143          RIGHT,
144    
145          /**
146           * The keyboard key that removes the card at inlay position 1 assuming
147           * that there is a card there
148           */
149          REMOVE_INLAY_POS_ONE,
150    
151          /**
152           * The keyboard key that removes the card at inlay position 2 assuming
153           * that there is a card there
154           */
155          REMOVE_INLAY_POS_TWO,
156    
157          /** The keyboard key that buzzes in Player 1 */
158          PLAYER1_BUZZ,
159    
160          /** The keyboard key that buzzes in Player 2 */
161          PLAYER2_BUZZ,
162    
163          /** The keyboard key that buzzes in Player 3 */
164          PLAYER3_BUZZ,
165    
166          /** The keyboard key that buzzes in Player 4 */
167          PLAYER4_BUZZ,
168       }
169    
170       /**
171        * Constructs KeyBindings with previously saved key binds.
172        */
173       public KeyBindings()
174       {
175          // CREATE a new HashMap and set it to keyBindings
176          keyBindings = new HashMap<KEY_BIND, String>();
177    
178          // ADD the hint key to keyBindings
179          keyBindings.put(KEY_BIND.HINT, kKeyHint);
180    
181          // ADD the pause key to keyBindings
182          keyBindings.put(KEY_BIND.PAUSE, kKeyPause);
183    
184          // ADD the unpause key to keyBindings; it is the same key as pause
185          keyBindings.put(KEY_BIND.UNPAUSE, kKeyUnpause);
186    
187          // ADD the three more key to keyBindings
188          keyBindings.put(KEY_BIND.THREE_MORE, kKeyThreeMore);
189    
190          // ADD the select key to keyBindings
191          keyBindings.put(KEY_BIND.SELECT, kKeySelect);
192    
193          // ADD the up key to keyBindings
194          keyBindings.put(KEY_BIND.UP, kKeyUp);
195    
196          // ADD the down key to keyBindings
197          keyBindings.put(KEY_BIND.DOWN, kKeyDown);
198    
199          // ADD the left key to keyBindings
200          keyBindings.put(KEY_BIND.LEFT, kKeyLeft);
201    
202          // ADD the right key to keyBindings
203          keyBindings.put(KEY_BIND.RIGHT, kKeyRight);
204    
205          // ADD the remove inlay key one to keyBindings
206          keyBindings.put(KEY_BIND.REMOVE_INLAY_POS_ONE, kKeyRemoveInlayOne);
207    
208          // ADD the remove inlay key two to keyBindings
209          keyBindings.put(KEY_BIND.REMOVE_INLAY_POS_TWO, kKeyRemoveInlayTwo);
210    
211          // ADD the player one buzz key to keyBindings
212          keyBindings.put(KEY_BIND.PLAYER1_BUZZ, kKeyBuzzOne);
213    
214          // ADD the player two buzz key to keyBindings
215          keyBindings.put(KEY_BIND.PLAYER2_BUZZ, kKeyBuzzTwo);
216    
217          // ADD the player three buzz key to keyBindings
218          keyBindings.put(KEY_BIND.PLAYER3_BUZZ, kKeyBuzzThree);
219    
220          // ADD the player four buzz key to keyBindings
221          keyBindings.put(KEY_BIND.PLAYER4_BUZZ, kKeyBuzzFour);
222       }
223    
224       /**
225        * Gets a specific key binding for a specific action
226        * 
227        * @param keyBinding
228        *           The keyboard binding for a specific action.
229        * @return The string value representing the keyboard binding.
230        */
231       public String getKeyBinding(KEY_BIND keyBinding)
232       {
233          // RETURN keyBindins' get with keyBinding
234          return (String) keyBindings.get(keyBinding);
235       }
236    
237       /**
238        * Sets and persist all the key binding in KEY_BIND with their key
239        * 
240        * @param keyMap
241        *           The HashMap of all the keyboard binding and there keys.
242        */
243       public void setKeyBindings(HashMap<KEY_BIND, String> keyMap)
244       {
245          // SET the keyBindings to keyMap
246          keyBindings = keyMap;
247       }
248    }
