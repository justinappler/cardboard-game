016     *      <p>Displays the state of the CardBoard game through 
017     *      a console. It creates a list of cards with a number to differentiate 
018     *      each card and then displays it's characteristics. It also creates

022     *      player selects a card on the board it will move to the Inlay and vice  
023     *      versa. The game will display if after selecting three cards a correct  
024     *      set was created.</p> 
025     * 
026     *      <b>Operations:</b> 
027     *      <ul> 
028     *      <li>TextUI displays the game state in a console.</li> 
029     *      <li>TextUI can prompt the user for his or her name and return it.</li> 
030     *      <li>TextUI can display error and explanation messages.</li> 
031     *      </ul>

033     *      @author         Kyle Williamson

139            /** 
140             *      Starts a new game and draws the first board. It then cycles through 
141             *      until the user enter's a 'q' character. Takes input and checks to 
142             *      see if it is a number and exists on the board or the inlay, otherwise 
143             *      calls gameChoice() method with the input. 
144             */ 
145             private void display() 
146             { 
147                 String line; 
148                 char option = 'a'; 
149                 SetsBoard board; 
150     
151                 System.out.println("Card Board"); 
152                 System.out.println("board won't display new time and scores until "
153                       + "user selects a card, refreshes, or time" 
154                       + " runs out"); 
155                 //Start the game type the user wants to play 
156                 preferences(); 
157                 //Print out the available options 
158                 options(); 
159                 //Draw a text version of the game 
160                 draw(); 
161                 // While player has not entered 'q' 
162                 do 
163                 { 
164                     int rows; 
165                     Natural nat; 
166                     //Get a current version of the game 
167                     board = game.getBoard(); 
168                     System.out.print(": "); 
169                     line = input.next(); 
170      
171                     //Set rows to the number of Rows being used in the game 
172                     rows = getRows(); 
173                     // For each row of the game 
174                     for(int row = 0; row < rows; row++) 
175                     { 
176                         //For each column of the game 
177                         for(int col = 0; col < cols; col++) 
178                         { 
179                             //Create a string out of the row and column 
180                             String finalString = Integer.toString(row) 
181                             + Integer.toString(col); 
182                             // If line is equal to a card number on the board 
183                             if(line.equals(finalString)) 
184                             { 
185                                 //Set the card as selected 
186                                 cardSelected(board.getCardFromBoard(new  
187                                 Natural(row), new Natural(col))); 
188                             } 
189                         } 
190                     } 
191                     //Set nat to the number of cards in the inlay 
192                     nat = board.getInlaySize(); 
193                     // For each card in the Inlay 
194                     for(int inlayIndex = 0; inlayIndex < nat.intValue(); inlayIndex++) 
195                     { 
196                         //Create a string out of the number of rows and inlay size 
197                         String finalString = Integer.toString(rows) 
198                         + Integer.toString(inlayIndex); 
199                         // If line is equal to a card number in the inlay 
200                         if(line.equals(finalString)) 
201                         { 
202                             //Set the card as selected 
203                             cardSelected(board.getInlayCard(new Natural(inlayIndex))); 
204                         } 
205                      } 
206                    //Get the first char of the input string 
207                    option = line.charAt(0);  
208                    //Call gameChoice to check the input against the options 
209                    gameChoice(option); 
210                    }while(option != 'q'); 
211                 System.out.println("End of game"); 
212                 System.exit(0); 
213             }

215            /** 
216             *      The options for starting a new game of CardBoard. Current game 
217             *      types are solitaire mode, beginner mode, competitive mode, or 
218             *      multiplayer mode 
219             */ 
220             private void preferences() 
221             { 
222                 //Set readIn to a non-valid character in preferences 
223                 char readIn = 'a'; 
224                 String line; 
225                 System.out.println("starting a new game\nenter q to exit, " 
226                       + "s for solitaire mode, b for beginner mode, c for " 
227                       + "competitive mode, m for multiplayer mode"); 
228                 // While user has not entered 'q', 's', 'b', 'c', or 'm' 
229                 do 
230                 { 
231                      
232                     System.out.print(": "); 
233                     line = input.next(); 
234                     readIn = line.charAt(0); 
235                 }while(readIn != 'q' && readIn != 's' && readIn != 'b' 
236                    &&  readIn != 'c' && readIn != 'm'); 
237                 // If user chose a solitaire game 
238                 if(readIn == 's') 
239                 { 
240                     game.processAction(ACTION.NEW_SOLITAIRE_GAME); 
241                 } 
242                 // If user chose a beginner game 
243                 else if(readIn == 'b') 
244                 { 
245                      game.processAction(ACTION.NEW_BEGINNER_GAME); 
246                 } 
247                 // If user chose a competitive game 
248                 else if(readIn == 'c') 
249                 { 
250                     CardBoardPreferences pref = CardBoardPreferences.getInstance(); 
251                     System.out.print("Enter the time limit you desire" 
252                          + " [15 seconds(f), 30 seconds(t), 45 second(o)," 
253                               + " or unlimited(u)]"); 
254                     //Do while input character is not a valid time limit 
255                     do 
256                     { 
257                         System.out.print(":"); 
258                         //Read the first character of input 
259                         line = input.next(); 
260                         readIn = line.charAt(0); 
261                     }while(readIn != 'f' && readIn != 't' && readIn != 'o' 
262                         && readIn != 'u'); 
263                     //If the user chose a fifteen second game 
264                     if(readIn == 'f') 
265                     { 
266                         pref.setPreference(CardBoardPreferences.PREF.
267                             TIME_LIMIT, "TIME_15SEC"); 
268                     } 
269                     //If the user chose a thirty second game 
270                     else if(readIn == 't') 
271                     { 
272                         pref.setPreference(CardBoardPreferences.PREF.
273                             TIME_LIMIT, "TIME_30SEC"); 
274                     } 
275                     //If the user chose a forty-five second game 
276                     else if(readIn == 'o') 
277                     { 
278                         pref.setPreference(CardBoardPreferences.PREF.
279                             TIME_LIMIT, "TIME_45SEC"); 
280                     } 
281                     //If the user chose a unlimited time game 
282                     else if(readIn == 'u') 
283                     { 
284                         pref.setPreference(CardBoardPreferences.PREF.
285                             TIME_LIMIT, "TIME_UNLIMITED"); 
286                     } 
287                     game.processAction(ACTION.NEW_COMPETITIVE_GAME); 
288                 } 
289                 // If chose wants a multiplayer game 
290                 else if(readIn == 'm') 
291                 { 
292                     CardBoardPreferences pref = CardBoardPreferences.getInstance(); 
293                     System.out.print("Enter the number of players you desire" 
294                          + " (min 2, max 4):"); 
295                     //Do while input character is not a valid number of players 
296                     do 
297                     { 
298                         //Read the first character of input 
299                         line = input.next(); 
300                         readIn = line.charAt(0); 
301                     }while(readIn != '2' && readIn != '3' && readIn != '4'); 
302                     //If the user chose a 2 player game 
303                     if(readIn == '2') 
304                     { 
305                         pref.setPreference(CardBoardPreferences.PREF.PLAYERS, "TWO"); 
306                     } 
307                     //If the user chose a 3 player game 
308                     else if(readIn == '3') 
309                     { 
310                         pref.setPreference(CardBoardPreferences.PREF.PLAYERS, "THREE"); 
311                     } 
312                     //If the user chose a 4 player game 
313                     else if(readIn == '4') 
314                     { 
315                         pref.setPreference(CardBoardPreferences.PREF.PLAYERS, "FOUR"); 
316                     } 
317                     game.processAction(ACTION.NEW_MULTIPLAYER_GAME); 
318                 } 
319                 // If user chose to quit 
320                 else 
321                 { 
322                     System.out.println("End of game"); 
323                     System.exit(0); 
324                 } 
325             }

327            /** 
328             *      Draws the text version of the Card Board game. 
329             **/ 
330            private void draw() 
331            { 
332                int rows; 
333                SetsBoard board = game.getBoard(); 
334                //Get the timer, and the score for all possible players 
335                Timer time = game.getTimer(); 
336                Score scoreOne = game.getScore(PLAYER.PLAYER1); 
337                Score scoreTwo = game.getScore(PLAYER.PLAYER2); 
338                Score scoreThree = game.getScore(PLAYER.PLAYER3); 
339                Score scoreFour = game.getScore(PLAYER.PLAYER4); 
340                //Set rows to the number of Rows being used in the game 
341                rows = getRows(); 
342                     
343                //Print out the time, scores, and cards left 
344                System.out.println("Timer: " + time.getSeconds() + " seconds"); 
345                System.out.println("    Player 1: " + scoreOne.getScore()); 
346                //If there are at least two players 
347                if(scoreTwo != null) 
348                {
349                    System.out.println("    Player 2: " + scoreTwo.getScore()); 
350                }
351                //If there are at least three players 
352                if(scoreThree != null) 
353                {
354                    System.out.println("    Player 3: " + scoreThree.getScore()); 
355                }
356                //If there are four players 
357                if(scoreFour != null) 
358                {
359                    System.out.println("    Player 4: " + scoreFour.getScore()); 
360                }
361                System.out.println("Cards left: " + board.getCardsLeft()); 
362                //Print out the game board vertically for each row 
363                for(int rowNum = 0; rowNum < rows; rowNum++) 
364                { 
365                    // Print the game for each volume 
366                    for(int colNum = 0; colNum < cols; colNum++) 
367                    { 
368                        Card currentPos = board.getCardFromBoard(new 
369                            Natural(rowNum), new Natural(colNum)); 
370                        //If the card has previously been selected 
371                        if(currentPos == null) 
372                        { 
373                            System.out.println(rowNum + "" + colNum  
374                                    + "\tCard has been selected"); 
375                        } 
376                        //Else the card has not been selected 
377                        else 
378                        { 
379                            //Print out a cards characteristic 
380                            System.out.println(rowNum + "" + colNum + "\t"
381                                  +  currentPos.getShape() + "\t" 
382                                  +  currentPos.getColor() + "\t" + currentPos.getFill() 
383                                    + "\t" + currentPos.getNumber()); 
384                        } 
385                     } 
386                  } 
387                  //Print out the cards that have been selected 
388                  System.out.println("\n\nInlay:"); 
389                  // For each card in the Inlay 
390                  for(int tempInlaySize = 0; tempInlaySize < inlaySize; tempInlaySize++) 
391                  { 
392                      //Get the current card in the Inlay 
393                      Card currentPos = board.getInlayCard(new Natural(tempInlaySize)); 
394                      //If the card has been selected 
395                      if(currentPos == null) 
396                      { 
397                          System.out.println(rows + "" + tempInlaySize  
398                                  + "\tEmpty"); 
399                      } 
400                      //If the card exists on the Inlay 
401                      else 
402                      { 
403                          System.out.println(rows + "" + tempInlaySize + "\t"  
404                                  + currentPos.getShape() + "\t"   
405                                  + currentPos.getColor() + "\t" + currentPos.getFill() 
406                                  + "\t" + currentPos.getNumber()); 
407                      } 
408                  } 
409             }

436           /** 
437            *       Turns the enum ROWS, which changes depending on the number of  
438            *       cards in play, into an int representing row size. 
439            *       @return Number of rows 
440            **/ 
441            private int getRows() 
442            { 
443                int rows = 0; 
444                SetsBoard board = game.getBoard();    
445      
446                ROWS enumRow = board.getRowSize(); 
447                // If there are four rows 
448                if(enumRow == ROWS.FOUR) 
449                { 
450                    rows = fourRows; 
451                } 
452                // If there are five rows 
453                else if(enumRow == ROWS.FIVE) 
454                { 
455                    rows = fiveRows; 
456                } 
457                // If there are six rows 
458                else if(enumRow == ROWS.SIX) 
459                { 
460                    rows = sixRows; 
461                } 
462                // If there are seven rows 
463                else 
464                { 
465                    rows = sevenRows; 
466                }       
467                return rows;       
468            }

470           /** 
471            *       Runs through options for the game, selecting one if 
472            *       given a valid choice. 
473            *       @param userChoice the first character the user typed in 
474            **/ 
475            private void gameChoice(char userChoice) 
476            { 
477                // If the user chose to refresh 
478                if(userChoice == 'r') 
479                { 
480                    draw(); 
481                } 
482                // If the user chose to start a new game 
483                if(userChoice == 'n') 
484                { 
485                    preferences(); 
486                    draw(); 
487                } 
488                // If the user chose three more 
489                if(userChoice == 't') 
490                { 
491                    game.processAction(ACTION.THREE_MORE); 
492                    draw(); 
493                } 
494                // If user chose to pause 
495                if(userChoice == 'p') 
496                { 
497                    game.processAction(ACTION.PAUSE); 
498                    // Print out a loops number of empty lines 
499                    for(int blankLines = 0; blankLines < loops; blankLines++) 
500                    { 
501                        System.out.println(""); 
502                    } 
503                } 
504                // If user chose to unpause 
505                if(userChoice == 'u') 
506                { 
507                    game.processAction(ACTION.UNPAUSE); 
508                    draw(); 
509                } 
510                // If user chose a hint 
511                if(userChoice == 'h') 
512                { 
513                    game.processAction(ACTION.HINT); 
514                    draw(); 
515                } 
516                // If user chose to view the options 
517                if(userChoice == 'o') 
518                { 
519                    options(); 
520                }          
521                // Player 1 has buzzed in 
522                if(userChoice == '!') 
523                { 
524                    game.processAction(ACTION.P1_BUZZED_IN); 
525                }   
526                // Player 2 has buzzed in 
527                if(userChoice == '@') 
528                { 
529                    game.processAction(ACTION.P2_BUZZED_IN); 
530                }   
531                // Player 3 has buzzed in 
532                if(userChoice == '#') 
533                { 
534                    game.processAction(ACTION.P3_BUZZED_IN); 
535                }   
536                // Player 4 has buzzed in 
537                if(userChoice == '$') 
538                { 
539                    game.processAction(ACTION.P4_BUZZED_IN); 
540                }
541                // Hidden entry because the textUI is not connected to high scores
542                if(userChoice == '-') 
543                { 
544                    String username = getName();
545                    System.out.println(username + ":");
546                } 
547            }
