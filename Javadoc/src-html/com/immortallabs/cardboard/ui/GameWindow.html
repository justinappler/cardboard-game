413        private void highlightCard(CardGraphic card)
414        {
415            boolean done = false;
416            //FOR every row on the gameboard 
417            for (int row = 0; row < numberOfRows && !done; row++)
418            {
419                // FOR each column in the board
420                for (int col = 0; col < kColumns && !done; col++)
421                {
422                    // IF the card at that row and col is not NULL
423                    if (cards[row][col] != null)
424                    {
425                        // GET the card from the cards array 
426                        if (cards[row][col].getCard().equals(card.getCard()))
427                        {
428                            // ASSIGN card to the card clicked
429                            selectedRow = row;
430                            selectedCol = col;
431                            done = true;
432                        }
433                    }
434                }
435            }
436        }

438        /**
439         * Sets the selected card to the clicked card.
440         * 
441         * @param card
442         *            the card that was selected
443         */
444        public void cardClicked(CardGraphic card)
445        {
446            // SET selected to the card
447            this.selected = card;
448            int beforeInlaySize = 0, afterSize = 0;
449            // CREATE shouldRepaint boolean
450            boolean shouldRepaint = true;
451            // GET the current sets board
452            SetsBoard board = state.getBoard();
453            
454            //IF the card is in the inlay THEN
455            if (!board.isInInlay(card.getCard()))
456            {
457                highlightCard(card);
458                getRightCard();
459            }
460            // ANIMATE the card to the openInlaySlot
461            // card.animateTo(getOpenInLaySpot());
462            beforeInlaySize = board.getInlaySize().intValue();
463            // CALL map.CardClicked with card
464            map.cardClicked(card);
465            afterSize = board.getInlaySize().intValue();
466            
467            //IF the before inlaySize is not equal to 1 and the after size is
468            // equal to 0 THEN 
469            if ((beforeInlaySize == kInlaySize - 1) && afterSize == 0)
470            {
471                shouldRepaint = false;
472            }
473            // Repaint the screen
474            if (shouldRepaint && !hasMessage)
475            {
476                repaint();
477            }
478        }

526        /**
527         * Sets the selectedCOL and selectedRow to the card currently to 
528         * the left of the selected card
529         */
530        private void getLeftCard()
531        {
532            // CREATE card = null
533            SetsBoard board = state.getBoard();
534            boolean done = false;
535            int row = selectedRow;
536            int col = selectedCol - 1;
537            // FOR each row in the game board
538            do
539            {
540                //IF the selected card is about to go out of bounds 
541                if (row == 0 && col < 0)
542                {
543                    //PUT the selected card to the most bottom right card 
544                    row = this.kCardRows[board.getRowSize().ordinal()] - 1;
545                    col = this.kColumns - 1;
546                }
547                //ELSE IF if the col is about to go out of bounds 
548                else if (col < 0)
549                {
550                    //MAKE the row go down one  
551                    col = this.kColumns - 1;
552                    row--;
553                }
554                //ELSE IF the card at the row and col is not null 
555                if (cards[row][col] != null)
556                {
557                    //STOP trying to get the card 
558                    done = true;
559                }
560                //ELSE 
561                else
562                {
563                    //Move the selected card over a column 
564                    col--;
565                }
566            } while (!done);
567            selectedRow = row;
568            selectedCol = col;
569        }
570        /**
571         * Sets the selectedCOL and selectedRow to the card currently above
572         *  of the selected card
573         */
574        private void getUpCard()
575        {
576            boolean done = false;
577            int startRow = selectedRow; 
578            int row = selectedRow - 1;
579            int col = selectedCol;
580            // FOR each row in the game board
581            do
582            {
583                //IF the next card is out of bounds 
584                if (row < 0)
585                {
586                    //Don't move the selected card 
587                    row = startRow;
588                    done = true;
589                }
590                else
591                {
592                    //IF the next card above the current card is exists 
593                    if (cards[row][col] != null)
594                    {
595                        // STOP searching for the next card 
596                        done = true;
597                    }
598                    else
599                    {
600                       //Move the selector up a row 
601                       row -= 1;
602                    }
603                }
604            } while (!done);
605            selectedRow = row;
606            selectedCol = col;
607        }
608        /**
609         * Sets the selectedCOL and selectedRow to the card currently below
610         * the selected card
611         */
612        private void getDownCard()
613        {
614            // CREATE card = null
615            SetsBoard board = state.getBoard();
616            boolean done = false;
617            int startRow = selectedRow; 
618            int row = selectedRow + 1;
619            int col = selectedCol;
620            // FOR each row in the game board
621            do
622            {
623                //IF the next card is out of bounds 
624                if (row > this.kCardRows[board.getRowSize().ordinal()])
625                {
626                    //Stop searching for the next card 
627                    row = startRow;
628                    done = true;
629                }
630                //ELSE 
631                else
632                {
633                    //IF the next card exists 
634                    if (cards[row][col] != null)
635                    {
636                        //STOP searching for the card 
637                        done = true;
638                    }
639                    else
640                    {
641                            //Move the card down a row 
642                            row += 1;
643                    }
644                }
645            } while (!done);
646            selectedRow = row;
647            selectedCol = col;
648        }
649        /**
650         * Sets the selectedCOL and selectedRow to the card currently to 
651         * the right of the selected card
652         */
653        private void getRightCard()
654        {
655            // CREATE card = null
656            SetsBoard board = state.getBoard();
657            boolean done = false;
658            int row = selectedRow;
659            int col = selectedCol + 1;
660            // FOR each row in the game board
661            do
662            {
663                //IF the next card is going out of bounds 
664                if (row == kCardRows[board.getRowSize().ordinal()] - 1
665                        && col >= this.kColumns)
666                {
667                    //STOP searching for the card 
668                    row = 0;
669                    col = 0;
670                }
671                //IF the selector is at the most right card in the row 
672                else if (col >= kColumns)
673                {
674                    //Move the selector down a row 
675                    col = 0;
676                    row++;
677                }
678                //IF the next card exists then 
679                if (cards[row][col] != null)
680                {
681                    //STOP searching for the next card 
682                    done = true;
683                }
684                //ELSE 
685                else
686                {
687                    //Make the selector go to the right 
688                    col++;
689                }
690            } while (!done);
691            selectedRow = row;
692            selectedCol = col;
693        }

760        /**
761         * Checks to see if the mouse clicked on one of the inlay cards
762         * 
763         * @param x
764         *            the x location where the mouse was pressed
765         * @param y
766         *            the y location where the mouse was pressed
767         * @return an inlay card if the x and y is within the bounds of the card,
768         *         otherwise NULL
769         */
770        private CardGraphic checkInLay(int x, int y)
771        {
772            // CREATE card = null
773            CardGraphic card = null;
774            // IF the x and y is within the bounds of the first in lay card THEN
775            if ((x >= inlayX && x <= (inlayX + kCardWidth[scaleIndex]))
776                    && ((y >= inlayY[firstInlay]) && (y <= inlayY[firstInlay]
777                            + kCardHeight[scaleIndex])))
778            {
779                // IF the first inlay card exists THEN
780                if (inlay[firstInlay] != null)
781                {
782                    // ASSIGN card to be the first inlay card
783                    card = inlay[firstInlay];
784                }
785            }
786            // ELSE IF the x and y is within the bounds of the second in lay card
787            // THEN
788            else if ((x >= inlayX && x <= (inlayX + kCardWidth[scaleIndex]))
789                    && ((y >= inlayY[secondInlay]) && (y <= inlayY[secondInlay]
790                            + kCardHeight[scaleIndex])))
791            {
792                // IF the second inlay card exists THEN

804         * Checks to see if the mouse clicked on a card
805         * 
806         * @param x
807         *            the x location where the mouse was pressed
808         * @param y
809         *            the y location where the mouse was pressed
810         * @return a card if the x and y is within the bounds of the card, otherwise
811         *         NULL
812         */
813        private CardGraphic checkCards(int x, int y)
814        {
815            // CREATE card = null
816            CardGraphic card = null;
817            // CREATE cardX = the starting location of the first card on the board
818            int cardX = cardStartLoc.x;
819            // CREATE cardY = the starting location of the second card on the board
820            int cardY = cardStartLoc.y;
821            // FOR each row in the game board
822            for (int row = 0; row < numberOfRows; row++)
823            {
824                // FOR each column in the board
825                for (int col = 0; col < kColumns; col++)
826                {
827                    // IF the x and y is within the bounds of the card THEN
828                    if ((x >= cardX && x <= (cardX + kCardWidth[scaleIndex]))
829                            && ((y >= cardY) && (y <= cardY
830                                    + kCardHeight[scaleIndex])))
831                    {
832                        // IF the card is exists within the board THEN
833                        if (cards[row][col] != null)
834                        {
835                            // ASSIGN card to the card clicked
836                            card = cards[row][col];
837                        }
838                    }
839                    // ASSIGN cardX = to the xCoordinate of the next card
840                    cardX = cardX + kCardXOffset + kCardWidth[scaleIndex];
841                }
842                // REASSIGN cardX = to the starting x location of the first card
843                cardX = cardStartLoc.x;
844                // ASSIGN cardY = to the next row on the game board
845                cardY = cardY + kCardYOffset + kCardHeight[scaleIndex];
846            }
847            // RETURN card
848            return card;
849        }

1558            
1559            CardGraphic card;
1560            //IF the selectedROW is greater than the number of rows on the board
1561            //reset it to 0,0
1562            if(selectedRow >= this.kCardRows[state.getBoard().getRowSize().ordinal()])
1563            {
1564               selectedRow = 0; 
1565               selectedCol = 0; 
1566            }
1567            card = cards[selectedRow][selectedCol];
1568            //IF the selected card exists 
1569            if (card == null)
1570            {
1571                Card tempCard =
1572                        state.getBoard().getCardFromBoard(new Natural(selectedRow),
1573                            new Natural(selectedCol));
1574                //IF the card from the board exists

1576                {

1578                            new CardGraphic(tempCard, map,
1579                                    new Natural(selectedRow), new Natural(
1580                                            selectedCol), this, new Point(
1581                                            cardStartLoc.x, cardStartLoc.y));

1588            }
1589            //ELSE

1592                selected = card;
