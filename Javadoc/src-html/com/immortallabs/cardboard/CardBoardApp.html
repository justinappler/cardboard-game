035    public class CardBoardApp extends JApplet 
036    {
037       
038       /**
039        * Starts the CardBoard Game.  While the method accepts arguments per the
040        * Java Specification, no arguments are used.
041        * 
042        * @param args Arguments sent to CardBoardApp when the game is started. 
043        */
044       public static void main(String[] args) 
045       {
046          // A flag to mark if the application will be running in graphical mode.
047          boolean isGraphical = true;
048          
049          // A flag to mark if the application will be running in test mode.
050          boolean isNormal = true;
051    
052          // FOR EACH argument
053          for (String arg : args)
054          {
055             // IF the argument is "-text"
056             if (arg.equals("-text"))
057             {
058                // SET that the program will run in text mode.
059                isGraphical = false;
060             }
061    
062             // IF the argument is "-t"
063             else if (arg.equals("-t"))
064             {
065                // SET that the program will run in test mode.
066                isNormal = false;
067             }
068          }
069    
070          // IF the game is set to run graphically,
071          if (isGraphical)
072          {
073             //CREATE a Jframe 
074             javax.swing.JFrame app = new javax.swing.JFrame();
075             
076             //SET the frame to exit on close 
077             app.setDefaultCloseOperation(javax.swing.JFrame.EXIT_ON_CLOSE);
078             
079             //CREATE a game window 
080             GameWindow gw = new GameWindow(app,isNormal);
081             
082             // CREATE the menu bar and set it to the applet's menu bar  
083             app.setJMenuBar(gw.menu.getMenuBar());
084             
085             //ADD the game window to the frame 
086             app.add(gw); 
087             
088             //SET the size of the frame 
089             app.setSize(gw.kWindowWidth, gw.kWindowHeight);   
090             
091            //SET the visibility of the game window to true 
092             app.setVisible(true);             
093             
094             //CREATE the off screen image for the game window 
095             gw.makeImageBuffer();
096             
097             gw.repaint(); 
098             
099          }
100          // IF the game is not set to run graphically,
101          else
102          {
103             //CREATE a new GameState
104             GameState state = new GameState(isNormal);
105             
106             // CREATE a new TextUI.
107             TextUI window = new TextUI(state);
108          }
109       }
110       
111       /**
112        * Initializes the game window.
113        */
114       public void init() 
115       {
116          //CREATE an offscreen image for the window  
117          GameWindow gw = new GameWindow(this,true);
118          
119          //CREATE the menu bar and set it to the applet's menu bar  
120          setJMenuBar(gw.menu.getMenuBar());
121          
122          //ADD the game window to the applet's content pane 
123          getContentPane().add(gw);
124          
125          //SET the applet to be visible 
126          this.setVisible(true); 
127          
128          //SET the focus to be in the applet
129          this.requestFocus(); 
130          
131          //SET the visibility of the game window to true 
132          gw.setVisible(true);
133     
134          //CREATE the off screen image for the game window 
135          gw.makeImageBuffer();
136          
137    
138         } 
139
