002    /**
003     * The Card class contains all the necessary information to represent a 
004     * Cardboard game card. It contains enumerations and "getter" methods 
005     * for each attribute of the card.
006     * <br /><br />
007     * <b>Operations:</b>
008     * <ul>
009     * <li>The Card class represents a card from the game CardBoard.
010     * </ul>
011     * @author Michael Brooks
012     * @author ImmortalLabs CSC309 W'09

023    
024    public class Card implements Cloneable
025    {
026       /**
027        * The Shape enumeration describes the shape of the icons on the card.
028        */
029       public enum SHAPE
030       {
031          /** The squiggles shape **/     
032          SQUIGGLES, 
033          /** The diamonds shape **/
034          DIAMONDS, 
035          /** The ovals shape **/
036          OVALS
037       }
038       /**
039        * The Color enumeration describes the color of the icons on the card.
040        */
041       public enum COLOR
042       {
043           /** The color red **/
044          RED, 
045           /** The color purple **/
046          PURPLE, 
047           /** The color green **/      
048          GREEN
049       }
050       /**
051        * The Fill enumeration describes the fill type of the icons on the card.
052        */
053       public enum FILL
054       {
055          /** The solid fill type **/
056          SOLID, 
057          /** The solid striped type **/
058          STRIPED, 
059          /** The solid empty type **/
060          EMPTY
061       }
062       /**
063        * The Number enumeration describes the number of icons on the card.
064        */
065       public enum NUMBER
066       {
067          /** One icon on the card **/
068          ONE, 
069          /** Two icons on the card **/      
070          TWO, 
071          /** Three icons on the card **/ 
072          THREE
073       }
074    
075       /**
076        * Contains the shape of the card's icons
077        */
078       private SHAPE shape;
079       /**
080        * Contains the color of the card's icons
081        */   
082       private COLOR color;
083       /**
084        * Contains the fill type of the card's icons
085        */   
086       private FILL fill;
087       /**
088        * Contains the number of icons on the card
089        */   
090       private NUMBER number;
091       
092       /**
093        * Creates a card with the shape, color, fill and number passed.
094        * @param   shape the shape of the icons on the card
095        * @param   color the color of the icons on the card
096        * @param   fill  the fill type of the icons on the card
097        * @param   number   the number of icons on the card
098        */
099       public Card(SHAPE shape, COLOR color, FILL fill, NUMBER number)
100       {
101          //SET this card's shape
102          this.shape = shape;
103          //SET this card's color
104          this.color = color;
105          //SET this card's fill
106          this.fill = fill;
107          //SET this card's number
108          this.number = number;
109       }
110    
111       /**
112        * This method returns the shape of the icons on the card.
113        * @return  Returns the Shape of the icons
114        */
115       public SHAPE getShape()
116       {
117          //RETURN shape
118          return shape;
119       }
120    
121       /**
122        * This method returns the color of the icons on the card.
123        * @return  Returns the Color of the icons
124        */
125       public COLOR getColor()
126       {
127          //RETURN color
128          return color;
129       }
130    
131       /**
132        * This method returns the fill type of the icons on the card.
133        * @return  Returns the Fill of the icons
134        */
135       public FILL getFill()
136       {
137          //RETURN fill
138          return fill;
139       }
140    
141       /**
142        * This method returns the number of icons on the card.
143        * @return  Returns the Number of icons
144        */
145       public NUMBER getNumber()
146       {
147          //RETURN number
148          return number;
149       }
150       
151       /**
152        * This method returns an identical card to this one.
153        * @return an identical copy of this card.
154        */
155       public Object clone()
156       {
157          Card clone = new Card(this.shape, this.color, this.fill, this.number);
158          return clone;     
159       }
160       
161       
162       /**
163        * Discovers if another object equals this card.
164        * @param obj the object being compared to.
165        * @return true if objects are equal, false otherwise.
166        */   
167       public boolean equals(Object obj)
168       {
169          // CREATE boolean retValue=false
170          boolean retValue = false;
171          // IF this card equals obj
172          if (this == obj)
173          {
174             // SET retValue to true
175             retValue = true;
176          }
177          // IF obj is not null and obj is a card
178          else if (!(obj == null) && obj instanceof Card)
179          {
180             // CREATE CARD other = obj casted to Card
181             Card other = (Card)obj;
182             
183             // IF other's shape equals this shape and 
184             // other's color equals this color and
185             // other's fill equals this fill and 
186             // other's number equals this number 
187             if (other.getShape()  == this.shape 
188                 && other.getColor() == this.color
189                 && other.getFill() == this.fill
190                 && other.getNumber() == this.number)
191             {
192                // retValue=true
193                retValue = true;
194             }
195                
196          }
197          
198          //RETURN retValue
199          return retValue;
200       }   
201    }
