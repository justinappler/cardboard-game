001    package com.immortallabs.cardboard.game; 
002     
003    import java.util.ArrayList; 
004    import java.util.Collections; 
005    import java.util.Stack; 
006     
007    import com.immortallabs.cardboard.game.Card.COLOR; 
008     
009    import edu.profdalbey.Natural; 
010     
011    /** 
012     *      <p>Models a stack of CardBoard cards in a 
013     *      random order. You can deal card off the 
014     *      deck and ask the deck if it's empty.</p> 
015     * 
016     *      <b>Operations:</b> 
017     *      <ul> 
018     *      <li>Cards can be dealt off the deck.</li> 
019     *      <li>The deck can be asked whether or not it is empty.</li> 
020     *      </ul> 
021     *      @author         Kyle Williamson 
022     *      @author         Formerly Sky Eckstrom 
023     *      @author         ImmortalLabs CSC308 W09 
024     *      @version        1.4 
025     *      @version        1/18/09 
026     * 
027     *      @see Card 
028     * 
029     *      @history 
030     *              - 2/18/09  1.4  Added private method to deal with duplicate code 
031     *              - 1/18/09  1.3  Wrote out code from pseudocode 
032     *              - 1/13/09  1.2  Switched to using the Collection's shuffle 
033     *              - 1/13/09  1.2  Added methods to create cards and add them to the deck 
034     *              - 1/12/09  1.1  Control of Deck given to Kyle Williamson 
035     *              - 12/04/08 1.1  Added Version History & Pseudocode<br> 
036     *              - 11/30/08 1.0  Inital Class Skeleton<br> 
037     **/ 
038      
039    public class Deck 
040    { 
041             /** 
042             *      A stack of cards modeling a deck 
043             *      that can be accessed in order. 
044             **/ 
045           private Stack<Card> deck; 
046     
047            /** 
048             *      Size of a normal deck 
049             **/ 
050           private final int normalSize = 81; 
051     
052            /** 
053             *      Size of a beginner deck 
054             **/ 
055           private final int beginSize = 27; 
056            
057           /** 
058            *       Number of cards in the current Deck 
059            */ 
060           private Natural cardsInDeck; 
061     
062             /** 
063             *      Instantiates a deck filled with 
064             *      every available card in a random 
065             *      order 
066             * 
067             *      @param gameType  True for a normal Deck, false  
068             *                       for a beginner deck 
069             *      @param shuffle   flag for if the deck will be shuffled 
070             **/ 
071            public Deck(boolean gameType, boolean shuffle) 
072            { 
073                    /* Pile of cards representing the deck */ 
074                    deck = new Stack<Card>(); 
075                    /* If the game is going to use a normal deck */ 
076                    if(gameType) 
077                    { 
078                      cardsInDeck = new Natural(normalSize); 
079                        createDeck(true); 
080                    } 
081                    /* If the game is going to use a beginner deck */ 
082                    else 
083                    { 
084                      cardsInDeck = new Natural(beginSize); 
085                        createDeck(false); 
086                    } 
087                    /* If the deck is going to be shuffles */ 
088                    if(shuffle) 
089                    { 
090                            Collections.shuffle(deck); 
091                    } 
092            } 
093             
094             
095            /** 
096             * Creates the Cards used in a deck. If a normal size deck it has 81 cards 
097             * and 3 different Numbers, Colors, Shapes, and Fills. If it's a beginner 
098             * deck it has 27 cards, 3 different Numbers, Shapes, and Fills but only one 
099             * Color 
100             * @param normDeck A boolean to determine if a normal size deck is created 
101             * or a beginner deck. 
102             */ 
103            private void createDeck(boolean normDeck) 
104            { 
105             ArrayList<Card> placeInOrder = new ArrayList<Card>(); 
106             int index; 
107             /* For every card's number */ 
108                for(Card.NUMBER num : Card.NUMBER.values()) 
109                { 
110                    /* If the deck is normal size */ 
111                    if(normDeck) 
112                    { 
113                        /* For every card's color */ 
114                        for(Card.COLOR col : Card.COLOR.values()) 
115                        { 
116                         //Add created cards to the ArrayList 
117                         placeInOrder.addAll(addNumCol(num, col)); 
118                        } 
119                    } 
120                    //Else the deck is a beginner size 
121                    else 
122                    { 
123                      //Using only the color Red 
124                        Card.COLOR col = COLOR.RED; 
125                        //Add created cards to the ArrayList 
126                        placeInOrder.addAll(addNumCol(num, col)); 
127                    } 
128                } 
129                //For the size of the deck minus 1 to account for index errors, 
130                //switch the order of cards in the deck. Required because of the 
131                //order of the enums in the Card class. 
132                for(index = cardsInDeck.intValue()-1; index >= 0; index--) 
133                { 
134                       deck.push(placeInOrder.get(index)); 
135                } 
136            } 
137             
138            /** 
139             * A private method used by createDeck to remove duplicate code used in  
140             * the creation of both normal decks and beginner decks. Automatically 
141             * adds the correct shapes and fills to the deck. 
142             * @param num The number of shapes to be added in the Cards. 
143             * @param col The color of the shapes to be added in the Cards. 
144             * @return A set of the cards used in the Deck. 
145             */ 
146            private ArrayList<Card> addNumCol(Card.NUMBER num, Card.COLOR col) 
147            { 
148             ArrayList<Card> createSetOfCards = new ArrayList<Card>(); 
149             /* For every card's shape */ 
150                for(Card.SHAPE sha : Card.SHAPE.values()) 
151                { 
152                    /* For every card's fill */ 
153                    for(Card.FILL fil : Card.FILL.values()) 
154                    { 
155                      //Create a new Card with the prescribed characteristics 
156                        Card tempCard = new Card(sha, col, fil, num); 
157                        //Add the created card to the ArrayList 
158                        createSetOfCards.add(tempCard); 
159                    }    
160                } 
161                return createSetOfCards; 
162            } 
163     
164            /** 
165             *      Deals the top card off the deck. Returns 
166             *      null if there are no more cards in the 
167             *      deck. 
168             * 
169             *      @return          the top card of the Deck or 
170             *                       null if the deck is empty. 
171             **/ 
172            public Card deal() 
173            { 
174                    /* If the deck is empty */ 
175                    if(isEmpty()) 
176                    { 
177                            return null; 
178                    } 
179                    cardsInDeck.decrement(); 
180                    return deck.pop(); 
181            } 
182     
183            /** 
184             *      Returns whether or not the deck is empty. 
185             * 
186             *      @return         TRUE if the deck is empty, 
187             *                      FALSE otherwise. 
188             **/ 
189            public boolean isEmpty() 
190            { 
191                    return deck.empty(); 
192            } 
193             
194            /** 
195             * Gives the number of cards that are left in the deck. 
196             * @return   A natural of the number of cards 
197             */ 
198            public Natural getCardsLeft() 
199            { 
200             return cardsInDeck; 
201            } 
202    }
