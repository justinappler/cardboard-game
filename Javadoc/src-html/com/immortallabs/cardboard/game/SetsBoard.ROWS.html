229        /**
230         * If a set is selected and removed from the game board: Adds three new
231         * cards in old cards positions. If there is not a set on the board: Adds a
232         * row and three more Cards to the game board.
233         */
234        public void addThreeMore()
235        {
236    
237            // SET isSpot to false
238            boolean isSpot = false;
239    
240            // SET first, second, and third to -1
241            int first = -1, second = -1, third = -1;
242    
243            // IF size of inlay == max size of inlay
244            if (getInlaySize().intValue() == kInlaySize)
245            {
246                // FOR iterator=0 to size of cardList
247                for (int iterator = 0; iterator < cardList.size(); iterator++)
248                {
249                    // IF current card == null
250                    if (cardList.get(iterator) == null)
251                    {
252                        // IF first < 0
253                        if (first < 0)
254                        {
255                            // first = iterator
256                            first = iterator;
257                        }
258                        // ELSE IF second < 0
259                        else if (second < 0)
260                        {
261                            // second = iterator
262                            second = iterator;
263                        }
264                        // ELSE IF third < 0
265                        else if (third < 0)
266                        {
267                            // third = iterator
268                            third = iterator;
269                        }
270                        // END IF
271    
272                        // SET isSpot to true
273                        isSpot = true;
274                    }
275                    // END IF
276                }
277                // END FOR
278    
279                // IF isSpot and numRows > 0
280                if (isSpot && numRows.ordinal() > 0)
281                { 
282                    // SET origSize to the index in the cardList of the card
283                    int origSize = (numRows.ordinal() + kInlaySize) * kInlaySize;
284    
285                    // SET index to origSize
286                    int index = origSize;
287    
288                    // SET breakOut to false
289                    boolean breakOut = false;
290    
291                    // FOR the amount of cards on the board
292                    for (int iterator = 0; iterator 
293                            < (numRows.ordinal() + kInlaySize) * kInlaySize; 
294                                    iterator++)
295                    {
296                        // SET card to the current card
297                        Card card = cardList.get(iterator);
298    
299                        // IF card is null
300                        if (card == null)
301                        {
302                            // FOR the amount of cards on the board
303                            for (; index < cardList.size() && !breakOut; index++)
304                            {
305                                // SET C2 to current card
306                                Card c2 = cardList.get(index);
307    
308                                // IF c2 is not null
309                                if (c2 != null)
310                                {
311                                    // REPLACE the card at the current spot with c2
312                                    cardList.set(iterator, c2);
313    
314                                    // SET breakOut to be true to stop
315                                    breakOut = true;
316                                }
317                                // END IF
318                            }
319    
320                            // SET breakOut back to false for next time through
321                            breakOut = false;
322                        }
323                        // END FOR
324                    }
325                    // END FOR
326    
327                    // REMOVE the last three spots in cardlist
328                    cardList.remove(origSize);
329                    cardList.remove(origSize);
330                    cardList.remove(origSize);
331    
332                    // IF numRows == FIVE
333                    if (numRows == ROWS.FIVE)
334                    {
335                        // SET numRows to four
336                        numRows = ROWS.FOUR;
337                    }
338                    // ELSE IF numRows is six
339                    else if (numRows == ROWS.SIX)
340                    {
341                        // SET numRows to five
342                        numRows = ROWS.FIVE;
343                    }
344                    // ELSE IF numRow is seven
345                    else if (numRows == ROWS.SEVEN)
346                    {
347                        // SET numRow to six
348                        numRows = ROWS.SIX;
349                    }
350                    // END IF
351                }
352    
353                // IF isSpot == true
354                else if (isSpot)
355                {      
356                    // IF gameDeck is not empty
357                    if (!gameDeck.isEmpty())
358                    {
359                        // SET cardList to a new card at position of first
360                        cardList.set(first, gameDeck.deal());
361                    }
362                    // END IF
363    
364                    // IF gameDeck is not empty
365                    if (!gameDeck.isEmpty())
366                    {
367                        // SET cardList to a new card at position of second
368                        cardList.set(second, gameDeck.deal());
369                    }
370                    // END IF
371    
372                    // IF gameDeck is not empty
373                    if (!gameDeck.isEmpty())
374                    {
375                        // SET cardList to a new card at position of third
376                        cardList.set(third, gameDeck.deal());
377                    }
378                    // END IF
379    
380                    // IF gameDeck is not empty
381                    if (gameDeck.isEmpty())
382                    {
383                        // SET deckEmpty to true
384                        deckEmpty = true;
385                    }
386                    // END IF
387                }
388                // END IF
389            }
390            // ELSE IF board does not contain a set.
391            else if (!boardContainsSet())
392            {
393                // IF numRows == FOUR
394                if (!gameDeck.isEmpty() && numRows == ROWS.FOUR)
395                {
396                    // numRows = FIVE
397                    numRows = ROWS.FIVE;
398                }
399                // ELSE IF numRows == FIVE
400                else if (!gameDeck.isEmpty() && numRows == ROWS.FIVE)
401                {
402                    // numRows = SIX
403                    numRows = ROWS.SIX;
404                }
405                // ELSE IF numRows == SIX
406                else if (!gameDeck.isEmpty() && numRows == ROWS.SIX)
407                {
408                    // numRows = SEVEN
409                    numRows = ROWS.SEVEN;
410                }
411                // END IF
412    
413                // FOR iterator=0 to size of a set
414                for (int iterator = 0, index = cardList.size(); 
415                    iterator < kInlaySize; iterator++)
416                {
417                    // IF gameDeck is not empty
418                    if (!gameDeck.isEmpty())
419                    {
420                        // ADD a card to cardList at index position
421                        cardList.add(index++, gameDeck.deal());
422                    }
423                    // ELSE
424                    else
425                    {
426                        // SET deckEmpty to true
427                        deckEmpty = true;
428                    }
429                    // END IF
430                }
431                // END FOR
432            }
433            // END IF
434    
435            // CALL makeSet
436            makeSet();
437        }

506        /**
507         * Verifies that three given Cards constitute a set
508         * 
509         * @param card1
510         *            first Card
511         * @param card2
512         *            second Card
513         * @param card3
514         *            third Card
515         * @return true if the three Cards constitute a set false if the three Cards
516         *         do not constitute a set
517         */
518        public boolean checkForSet(Card card1, Card card2, Card card3)
519        {
520            // SET isSet to true
521            boolean isSet = true;
522    
523            // IF card1 and card2 and card3 are null
524            if (card1 == null || card2 == null || card3 == null)
525            {
526                // SET isSet to false
527                isSet = false;
528            }
529            // ELSE IF card1 == card2 and card2 == card3 and card1 == card3
530            else if (card1.equals(card2) && card2.equals(card3)
531                    && card1.equals(card3))
532            {
533                // SET isSet to false
534                isSet = false;
535            }
536    
537            // ELSE
538            else
539            {
540                // SET isSet to the AND of the return value of isFeatureSet with
541                // each cards color
542                isSet &=
543                        isFeatureSet(card1.getColor(), card2.getColor(), card3
544                                .getColor());
545                // SET isSet to the AND of the return value of isFeatureSet with
546                // each cards shape
547                isSet &=
548                        isFeatureSet(card1.getShape(), card2.getShape(), card3
549                                .getShape());
550                // SET isSet to the AND of the return value of isFeatureSet with
551                // each cards fill
552                isSet &=
553                        isFeatureSet(card1.getFill(), card2.getFill(), card3
554                                .getFill());
555                // SET isSet to the AND of the return value of isFeatureSet with
556                // each cards number
557                isSet &=
558                        isFeatureSet(card1.getNumber(), card2.getNumber(), card3
559                                .getNumber());
560            }
561            // END IF
562    
563            // RETURN isSet
564            return isSet;
565        }

567        /**
568         * Compares three enumerations and return true if they are all different or
569         * all the same.
570         * 
571         * @param e1
572         *            the first enum
573         * @param e2
574         *            the second enum
575         * @param e3
576         *            the third enum
577         * @return true if all enumerations are the same or all different false
578         *         otherwise
579         */
580    
581        @SuppressWarnings("unchecked")
582        private boolean isFeatureSet(Enum e1, Enum e2, Enum e3)
583        {
584            // SET isFeatureSet to false
585            boolean isFeatureSet = false;
586    
587            // IF e1 and e2 and e3 are not all equal
588            if (e1 != e2 && e1 != e3 && e2 != e3)
589            {
590                // SET isFeatureSet to true
591                isFeatureSet = true;
592            }
593            // END IF
594    
595            // IF e1 and e2 and e3 are all equal
596            if (e1 == e2 && e2 == e3)
597            {
598                // SET isFeatureSet to true
599                isFeatureSet = true;
600            }
601    
602            // RETURN isFeatureSet
603            return isFeatureSet;
604        }

793         * @param isHint :
794         *            if true than card is being moved to in-lay due to using a
795         *            hint. if false than card was chosen by the player.
796         * @return : true if the move to the in-lay was a success false if the move
797         *         to the in-lay was a failure
798         */
799        public boolean moveCardToInlay(Card theCard, Boolean isHint)
800        {
801            // CREATE a boolean retValue = false
802            boolean retValue = false;
803    
804            // CREATE an index int
805            int index;
806    
807            // IF isHint equals false /*Card is being moved by user clicking Card*/
808            if (!isHint)
809            {
810                // SET index to return value of cardToIndex(theCard)
811                index = cardToIndex(theCard);
812    
813                // IF index >= 0 AND index < size of cardList
814                if (index >= 0 && index < cardList.size())
815                {
816                    // FOR size of board
817                    for (int iterator = 0; iterator < kInlaySize; iterator++)
818                    {
