001    package com.immortallabs.cardboard.game;
002    
003    /**
004     * <p>
005     * The CardBoardMessage class models any of the various messages displayed to
006     * the user.  While each message type has default text, the messages can be
007     * changed to suit the specific situation.
008     * </p>
009     * <b>Operations:</b>
010     * <ul>
011     * <li>The CardBoardMessage class change the text of the message
012     * <li>The CardBoardMessage class can provide the English text of a message
013     * </ul>
014     * 
015     * @author Justin C. Appler
016     * @author Immortal Labs, CSC309 W09
017     * @version 1.0
018     * @version March 8th, 2009
019     * @see GameState
020     * @see GameWindow
021     * @history - 03/08/09 1.0 Initial Class Design & Implementation<br>
022     */
023    public enum CardBoardMessage
024    {
025        /** The message displayed when a set is found */
026        CORRECT_SET("You found a set!"),
027        
028        /** The message displayed when an invalid set is found */
029        INVALID_SET("Not a valid set!"),
030        
031        /** The message displayed when the timer expires */
032        TIMER_EXPIRATION("You ran out of time."),
033        
034        /** 
035         * The message displayed when the timer expires and there was not set on 
036         * the board
037         */
038        NO_SET_TIMER_EXPIRATION("You ran out of time. There is no set on the board."),
039    
040        /** The message displayed when there is no set on the board */
041        NO_SET_ON_BOARD("There is no set on the board."),
042    
043        /** The message displayed when the game has been won */
044        GAME_WON("You won the game!"),
045    
046        /**  The message displayed when a hint cannot be used */
047        CANNOT_USE_HINT("You cannot use a hint after selecting a card."),
048        
049        /** 
050         * The message displayed when 3 more is selected 
051         * and there is a set on the board.
052         */
053        THREE_MORE_WHILE_SET("There is still a set on the board!"),
054    
055        /** The message displayed when a hint is attempted in competitive */
056        NO_HINT_IN_COMPETITIVE("You cannot use a hint in competitive mode!"),
057        
058        /** 
059         * The message displayed a card is clicked in multiplayer while
060         * no player is currently buzzed in
061         */
062        MUST_BUZZ_IN("Someone must buzz in first!"),
063        
064        /**
065         * The message shown when the user requests a valid or invalid set
066         * explanation.
067         */
068        EXPLANATION("");
069        
070        /** The text of the message */
071        String text = null;
072        
073        /**
074         * Constructs a cardboard message using the passed text
075         * @param text The text of the message
076         */
077        CardBoardMessage(String text)
078        {
079            // SET the passed text to the message's text
080            this.text = text;
081        }
082        
083        /**
084         * Sets the text of the message
085         * @param text The text of the message
086         */
087        public void setText(String text)
088        {
089            // SET the message text to the passed text
090            this.text = text;
091        }
092        
093        /**
094         * Gets the text of the message
095         * @return The text of the message
096         */
097        public String getText()
098        {
099            // RETURN the message text
100            return text;
101        }
102    }
