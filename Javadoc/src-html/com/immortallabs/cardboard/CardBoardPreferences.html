001    package com.immortallabs.cardboard;
002    
003    import java.util.HashMap;
004    
005    /**
006     * <p>
007     * The CardBoardPreferences class models all options for the game. If a player
008     * changes any options this class persists the change until next game is started
009     * </p>
010     * <b>Operations:</b>
011     * <ul>
012     * <li>The CardBoardPreferences class can get a current game preference
013     * <li>The CardBoardPreferences class can set a current game preference
014     * </ul>
015     * 
016     * @author Ryan Lange
017     * @author Formerly Thomas Dvornik
018     * @author RustyNail CSC308 F08
019     * @version 1.3
020     * @version 12/05/08
021     * @history - 02/20/09 1.2 Changed to a singleton class. - 12/02/08 1.1 Added
022     *          Version History & Pseudocode to CardBoardPreference(),
023     *          getPreferences(), setPreference()<br> 
024                - 12/03/08 1.2 Moved into correct package.<br> 
025     */
026    public class CardBoardPreferences
027    {
028        /** Represents limit limit of 15 seconds */
029        private String time15 = "TIME_15SEC";
030    
031        /** Represents limit limit of 30 seconds */
032        private String time30 = "TIME_30SEC";
033    
034        /** Represents limit limit of 45 seconds */
035        private String time45 = "TIME_45SEC";
036    
037        /** Represents an unlimited time limit */
038        private String timeUnlimited = "TIME_UNLIMITED";
039    
040        /** Represents a 1 player game */
041        private String players1 = "ONE";
042    
043        /** Represents a 2 player game */
044        private String players2 = "TWO";
045    
046        /** Represents a 3 player game */
047        private String players3 = "THREE";
048    
049        /** Represents a 4 player game */
050        private String players4 = "FOUR";
051    
052        /** Represents sound is on */
053        private String soundOn = "ON";
054    
055        /** Represents sound is off */
056        private String soundOff = "OFF";
057    
058        /** Holds each preference with is value */
059        private HashMap<PREF, String> prefMap;
060    
061        /** Holds the instance of CardBoardPreferenece */
062        private static CardBoardPreferences instance;
063    
064        /**
065         * The enumeration for different preferences
066         * 
067         * @author rslange
068         */
069        public enum PREF
070        {
071            /**
072             * The different preferences that a user can save for future games.
073             */
074            TIME_LIMIT, PLAYERS, SOUNDS;
075        }
076    
077        /**
078         * Constructs a new CardBoardPreferences
079         */
080        private CardBoardPreferences()
081        {
082            // INSTANTIATE A NEW HASHMAP
083            prefMap = new HashMap<PREF, String>();
084    
085            // LOAD prefMap with default settings
086            prefMap.put(PREF.TIME_LIMIT, time15);
087            prefMap.put(PREF.PLAYERS, players1);
088            prefMap.put(PREF.SOUNDS, soundOn);
089        }
090    
091        /**
092         * Gives the current state of the preferences
093         * 
094         * @return instance : The current instance of CardBoardPreferences
095         */
096        public static CardBoardPreferences getInstance()
097        {
098            // IF an instance has not been constructed yet
099            if (instance == null)
100            {
101                // CALL the constructer to get an instance of CardBoardPreferences
102                instance = new CardBoardPreferences();
103            }
104            // END IF
105    
106            // RETURN the newly created instance
107            return instance;
108        }
109    
110        /**
111         * @param pref
112         *            the desired enumeration
113         * @return The value of the desired enumeration
114         */
115        public String getPreference(PREF pref)
116        {
117            // RETURN the value of the given key from prefMap
118            return prefMap.get(pref);
119        }
120    
121        /**
122         * @param pref
123         *            The enumeration to modify
124         * @param value
125         *            The new value of the enumeration pref
126         */
127        public void setPreference(PREF pref, String value)
128        {
129            // SET validValue to false
130            boolean validValue = false;
131    
132            // IF pref is the TIME_LIMIT enum
133            if (pref == PREF.TIME_LIMIT)
134            {
135                // IF value is a valid time value
136                if (value.equals(time15) || value.equals(time30)
137                        || value.equals(time45) || value.equals(timeUnlimited))
138                {
139                    // SET validValue to true
140                    validValue = true;
141                }
142                // ELSE
143                else
144                {
145                    // PRINT error message
146                    System.err.println("Bad TIME_LIMIT: " + value);
147                }
148                // END IF
149            }
150            // ELSE IF pref is the PLAYERS enum
151            else if (pref == PREF.PLAYERS)
152            {
153                // IF value is a valid players value
154                if (value.equals(players1) || value.equals(players2)
155                        || value.equals(players3) || value.equals(players4))
156                {
157                    // SET validValue to true
158                    validValue = true;
159                }
160                // ELSE
161                else
162                {
163                    // PRINT error message
164                    System.err.println("Bad PLAYERS: " + value);
165                }
166                // END IF
167            }
168            // ELSE IF pref is the SOUNDS enum
169            else if (pref == PREF.SOUNDS)
170            {
171                // IF value is a valid value
172                if (value.equals(soundOn) || value.equals(soundOff))
173                {
174                    // SET validValue to true
175                    validValue = true;
176                }
177                // ELSE
178                else
179                {
180                    // PRINT error message
181                    System.err.println("Bad SOUND: " + value);
182                }
183                // END IF
184            }
185            // END IF
186    
187            // IF validValue is true
188            if (validValue)
189            {
190                // SET the given key pref to the given value in prefMap
191                prefMap.put(pref, value);
192            }
193            // END IF
194        }
195    }
